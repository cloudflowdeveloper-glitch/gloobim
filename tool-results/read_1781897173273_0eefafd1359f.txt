   755→function timeAgo(dateStr) {
   756→    if (!dateStr) return '';
   757→    const now = new Date();
   758→    const date = new Date(dateStr.replace(/-/g, '/'));
   759→    const diff = Math.floor((now - date) / 1000);
   760→    if (diff < 60) return 'just now';
   761→    if (diff < 3600) return Math.floor(diff / 60) + 'm';
   762→    if (diff < 86400) return Math.floor(diff / 3600) + 'h';
   763→    if (diff < 604800) return Math.floor(diff / 86400) + 'd';
   764→    if (diff < 2592000) return Math.floor(diff / 604800) + 'w';
   765→    return Math.floor(diff / 2592000) + 'mo';
   766→}
